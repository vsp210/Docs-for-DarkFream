  7class Request:
  8    """Класс для обработки HTTP-запросов.
  9
 10    Attributes:
 11        scope (dict): Контекст запроса.
 12        receive (callable): Функция для получения данных запроса.
 13        path (str): Путь запроса.
 14        method (str): Метод HTTP (например, GET, POST).
 15        headers (dict): Заголовки запроса.
 16        query_params (dict): Параметры запроса.
 17        body (bytes): Тело запроса.
 18    """
 19    def __init__(self, scope, receive):
 20        """Инициализация объекта Request.
 21
 22        Args:
 23            scope (dict): Контекст запроса.
 24            receive (callable): Функция для получения данных запроса.
 25        """
 26        self.scope = scope
 27        self.receive = receive
 28        self.path = self.scope.get('path', '/')
 29        self.method = self.scope.get('method', 'GET')
 30        self.headers = self.parse_headers(self.scope.get('headers', []))
 31        self.query_params = {}
 32        self.body = None
 33
 34    def parse_headers(self, headers):
 35        """Парсит заголовки запроса.
 36
 37        Args:
 38            headers (list): Список заголовков.
 39
 40        Returns:
 41            dict: Словарь заголовков с ключами в нижнем регистре.
 42        """
 43        return {k.decode('utf-8').lower(): v.decode('utf-8')
 44                for k, v in headers}
 45
 46    async def get_body(self):
 47        """Получает тело запроса.
 48
 49        Returns:
 50            bytes: Тело запроса.
 51        """
 52        if self.body is None:
 53            body = b''
 54            more_body = True
 55
 56            while more_body:
 57                message = await self.receive()
 58                body += message.get('body', b'')
 59                more_body = message.get('more_body', False)
 60
 61            self.body = body
 62
 63        return self.body
 64
 65    @property
 66    def content_type(self):
 67        """Возвращает тип контента из заголовков запроса."""
 68        return self.headers.get('content-type')
 69
 70    @property
 71    def content_length(self):
 72        """Возвращает длину контента из заголовков запроса."""
 73        return int(self.headers.get('content-length', 0))

 76    """Класс для обслуживания статических файлов.

 79        directory (str): Директория, из которой будут обслуживаться файлы.
 80    """
 81    def __init__(self, directory):
 82        """Инициализация объекта StaticFiles.
 83
 84        Args:
 85            directory (str): Директория для статических файлов.
 86        """
 87        self.directory = directory
 88
 89    def __call__(self, request):
 90        """Обрабатывает HTTP-запрос для статического файла.
 91
 92        Args:
 93            request (Request): Объект запроса.
 94
 95        Returns:
 96            tuple: Содержимое файла, статус ответа и тип контента.
 97        """
 98        file_path = request.path.replace('/static/', '', 1)
 99        full_path = os.path.join(self.directory, file_path)
100
101        try:
102            if os.path.exists(full_path) and os.path.isfile(full_path):
103                with open(full_path, 'rb') as f:
104                    content = f.read()
105
106                content_type = self.get_content_type(full_path)
107
108                return content, 200, content_type
109            else:
110                return "File not found", 404, 'text/plain'
111        except Exception as e:
112            print(f"Error serving static file: {e}")
113            return "Internal server error", 500, 'text/plain'

115    def get_content_type(self, path):
116        """Возвращает тип контента на основе расширения файла.
117
118        Args:
119            path (str): Путь к файлу.
120
121        Returns:
122            str: Тип контента.
123        """
124        ext = os.path.splitext(path)[1].lower()
125        content_types = {
126            '.css': 'text/css',
127            '.js': 'application/javascript',
128            '.jpg': 'image/jpeg',
129            '.jpeg': 'image/jpeg',
130            '.png': 'image/png',
131            '.gif': 'image/gif',
132            '.html': 'text/html',
133        }

193def calculator(input_str: str) -> float:
194    """Вычисляет результат математического выражения.
195
196    Args:
197        input_str (str): Строка, содержащая два числа и оператор.
198
199    Returns:
200        float: Результат вычисления.
201
202    Raises:
203        ValueError: Если строка не содержит два числа и оператора.
204        ZeroDivisionError: Если происходит деление на ноль.
205    """
206    numbers = input_str.split()
207
208    if len(numbers) != 3:
209        raise ValueError("Invalid input string. It should contain exactly two numbers and an operator.")
210
211    operator = numbers[1]
212    num1, num2 = float(numbers[0]), float(numbers[2])
213
214    if operator == "+":
215        result = num1 + num2
216    elif operator == "-":
217        result = num1 - num2
218    elif operator == "*":
219        result = num1 * num2
220    elif operator == "/":
221        if num2 == 0:
222            raise ZeroDivisionError("Cannot divide by zero!")
223        result = num1 / num2
224    elif operator == "**":
225        result = num1 ** num2
226    elif operator == "%":
227        result = num1 % num2
228    elif operator == "//":
229        if num2 == 0:
230            raise ZeroDivisionError("Cannot divide by zero!")
231        result = num1 // num2
232    else:
233        raise ValueError("Invalid operator. Supported operators are +, -, *, /, **, %, //.")
234
235    return result

238def send_email(sender_email, password, recipient_email, subject, body, host, port):
239    """Отправляет электронное письмо.
240
241    Args:
242        sender_email (str): Адрес электронной почты отправителя.
243        password (str): Пароль отправителя.
244        recipient_email (str): Адрес электронной почты получателя.
245        subject (str): Тема письма.
246        body (str): Текст письма.
247        host (str): Хост SMTP-сервера.
248        port (int): Порт SMTP-сервера.
249
250    Returns:
251        None
252    """
253    if not sender_email and not password:
254        return "Sender's email and password are required."
255
256    msg = EmailMessage()
257    msg.set_content(body)
258    msg['Subject'] = subject
259    msg['From'] = sender_email
260    msg['To'] = recipient_email
261
262    server = smtplib.SMTP(host, port)
263    server.starttls()
264    server.login(sender_email, password)
265    server.send_message(msg)
266    server.quit()

300class PluginManager:
301    """Класс для управления плагинами.

303    Attributes:

305        hooks (dict): Словарь хуков для расширения функциональности.
306    """
307    def __init__(self):
308        """Инициализация объекта PluginManager."""
309        self.plugins = {}
310        self.hooks = {}
311
312    def register_plugin(self, plugin_name: str, plugin_class: object) -> None:
313        """Регистрация нового плагина.
314
315        Args:
316            plugin_name (str): Имя плагина.
317            plugin_class (object): Класс плагина.
318
319        Raises:
320            ValueError: Если плагин с таким именем уже зарегистрирован.
321        """
322        if plugin_name in self.plugins:
323            raise ValueError(f"Plugin {plugin_name} already registered")
324        self.plugins[plugin_name] = plugin_class
325        print(f"Plugin '{plugin_name}' registered successfully")
326
327    def unregister_plugin(self, plugin_name: str) -> None:
328        """Удаление плагина.
329
330        Args:
331            plugin_name (str): Имя плагина для удаления.
332        """
333        if plugin_name in self.plugins:
334            del self.plugins[plugin_name]
335            print(f"Plugin '{plugin_name}' unregistered successfully")
336
337    def get_plugin(self, plugin_name: str) -> object:
338        """Получение плагина по имени.
339
340        Args:
341            plugin_name (str): Имя плагина.
342
343        Returns:
344            object: Класс плагина или None, если плагин не найден.
345        """
346        return self.plugins.get(plugin_name)
347
348    def register_hook(self, hook_name: str, callback: callable) -> None:
349        """Регистрация хука (точки расширения).
350
351        Args:
352            hook_name (str): Имя хука.
353            callback (callable): Функция обратного вызова для хука.
354        """
355        if hook_name not in self.hooks:
356            self.hooks[hook_name] = []
357        self.hooks[hook_name].append(callback)
358
359    def execute_hook(self, hook_name: str, *args, **kwargs) -> list:
360        """Выполнение всех callback-функций для определенного хука.
361
362        Args:
363            hook_name (str): Имя хука.
364            *args: Аргументы для передачи в функции обратного вызова.
365            **kwargs: Именованные аргументы для передачи в функции обратного вызова.
366
367        Returns:
368            list: Список результатов выполнения функций обратного вызова.
369        """
370        results = []
371        if hook_name in self.hooks:
372            for callback in self.hooks[hook_name]:
373                results.append(callback(*args, **kwargs))
374        return results

195    """Вычисляет результат математического выражения.
196
197    Args:
198        input_str (str): Строка, содержащая два числа и оператор.

200    Returns:
201        float: Результат вычисления.
202
203    Raises:
204        ValueError: Если строка не содержит два числа и оператора.
205        ZeroDivisionError: Если происходит деление на ноль.
206    """
207    numbers = input_str.split()
208
209    if len(numbers) != 3:
210        raise ValueError("Invalid input string. It should contain exactly two numbers and an operator.")
211
212    operator = numbers[1]
213    num1, num2 = float(numbers[0]), float(numbers[2])
214
215    if operator == "+":
216        result = num1 + num2
217    elif operator == "-":
218        result = num1 - num2
219    elif operator == "*":
220        result = num1 * num2
221    elif operator == "/":
222        if num2 == 0:
223            raise ZeroDivisionError("Cannot divide by zero!")
224        result = num1 / num2
225    elif operator == "**":
226        result = num1 ** num2
227    elif operator == "%":
228        result = num1 % num2
229    elif operator == "//":
230        if num2 == 0:
231            raise ZeroDivisionError("Cannot divide by zero!")
232        result = num1 // num2
233    else:
234        raise ValueError("Invalid operator. Supported operators are +, -, *, /, **, %, //.")
235
236    return result
